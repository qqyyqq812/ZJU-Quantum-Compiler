Python SDK Integration

仅需 3 行代码，将 AI Router 无缝接入您的 Qiskit 工作流，替换传统 transpile。

quickstart.py 
# 1. 导入 AI 路由器
from src.compiler.pass_manager import compile_with_ai
from qiskit import QuantumCircuit
from qiskit.transpiler import CouplingMap

# 2. 准备电路与目标拓扑
qc = QuantumCircuit.from_qasm_file("algorithms/qft_5.qasm")
cmap = CouplingMap.from_line(5)

# 3. 💥 一键接管 — 替代 transpile(qc, coupling_map=cmap)
optimized = compile_with_ai(qc, cmap, model_path="models/router_v3_linear_5.pt")
print(f"Done! Depth: {optimized.depth()}")
