# 1. 导入 V9 核心组件
from src.compiler.inference_v8 import load_policy, compile_multi_trial
from src.benchmarks.topologies import get_topology

# 2. 加载模型（自动探测 obs_dim）与 20Q 真实拓扑
policy, obs_dim = load_policy("models/v9_tokyo20/v7_ibm_tokyo_best.pt")

# 3. 💥 启动 MTx100 推理引擎 (Numpy 零拷贝)
optimized = compile_multi_trial(qc, policy, cmap, n_trials=100)
print(f"Done! Required SWAPs: {optimized.swaps}")
